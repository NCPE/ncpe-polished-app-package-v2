const DATA_URL = "data/crossovers.csv"; let allRows = []; const searchInput = document.getElementById("searchInput"); const clearBtn = document.getElementById("clearBtn"); const brandFilter = document.getElementById("brandFilter"); const typeFilter = document.getElementById("typeFilter"); const confidenceFilter = document.getElementById("confidenceFilter"); const results = document.getElementById("results"); const resultCount = document.getElementById("resultCount"); const quotePanel = document.getElementById("quotePanel"); const closeQuote = document.getElementById("closeQuote"); const quoteForm = document.getElementById("quoteForm"); const formStatus = document.getElementById("formStatus"); const submitQuote = document.getElementById("submitQuote"); const selectedCrossover = document.getElementById("selectedCrossover"); const selectedBarmesa = document.getElementById("selectedBarmesa"); const selectedCompetitor = document.getElementById("selectedCompetitor"); function parseCSV(text) { const rows = []; let row = [], value = "", inQuotes = false; for (let i = 0; i < text.length; i++) { const char = text[i]; const next = text[i + 1]; if (char === '"' && inQuotes && next === '"') { value += '"'; i++; } else if (char === '"') { inQuotes = !inQuotes; } else if (char === "," && !inQuotes) { row.push(value); value = ""; } else if ((char === "\n" || char === "\r") && !inQuotes) { if (value || row.length) { row.push(value); rows.push(row); row = []; value = ""; } if (char === "\r" && next === "\n") i++; } else { value += char; } } if (value || row.length) { row.push(value); rows.push(row); } const headers = rows.shift().map(h => h.trim()); return rows .filter(r => r.some(cell => String(cell).trim() !== "")) .map(r => Object.fromEntries(headers.map((h, i) => [h, (r[i] || "").trim()]))); } function normalize(value) { return String(value || "").toLowerCase().replace(/[^a-z0-9.]+/g, " ").trim(); } function fillFilters() { const brands = [...new Set(allRows.map(r => r.competitor_brand).filter(Boolean))].sort(); const types = [...new Set(allRows.map(r => r.pump_type).filter(Boolean))].sort(); brands.forEach(brand => { const option = document.createElement("option"); option.value = brand; option.textContent = brand; brandFilter.appendChild(option); }); types.forEach(type => { const option = document.createElement("option"); option.value = type; option.textContent = type; typeFilter.appendChild(option); }); } function rowText(row) { return normalize([ row.competitor_brand, row.competitor_model, row.pump_type, row.hp, row.voltage, row.phase, row.discharge_size, row.gpm, row.head_ft, row.application, row.barmesa_model, row.barmesa_series, row.verify_notes ].join(" ")); } function scoreRow(row, query) { if (!query) return 1; const q = normalize(query); const terms = q.split(" ").filter(Boolean); const text = rowText(row); let score = 0; if (normalize(row.competitor_model) === q) score += 100; if (normalize(row.competitor_model).includes(q)) score += 60; if (normalize(row.competitor_brand).includes(q)) score += 30; if (normalize(row.barmesa_model).includes(q)) score += 30; terms.forEach(term => { if (text.includes(term)) score += 8; }); return score; } function getFilteredRows() { const query = searchInput.value; const brand = brandFilter.value; const type = typeFilter.value; const confidence = confidenceFilter.value; return allRows .map(row => ({ row, score: scoreRow(row, query) })) .filter(item => { const row = item.row; return item.score > 0 && (!brand || row.competitor_brand === brand) && (!type || row.pump_type === type) && (!confidence || row.confidence === confidence); }) .sort((a, b) => b.score - a.score) .map(item => item.row); } function render() { const rows = getFilteredRows(); results.innerHTML = ""; resultCount.textContent = `${rows.length} crossover result${rows.length === 1 ? "" : "s"} found`; if (!rows.length) { results.innerHTML = `

No exact result found

Try searching partial model numbers, brand names, HP, discharge size, or application. You can still request a quote for manual review.

`; return; } rows.slice(0, 60).forEach(row => { const card = document.createElement("article"); card.className = "card"; card.innerHTML = `

${row.competitor_brand || "Competitor"} ${row.competitor_model || ""}

${row.confidence || "Medium"} Confidence
Recommended Barmesa equivalent
${row.barmesa_model || "Review Required"}
${row.barmesa_series || ""} Series
Type:
${row.pump_type || "-"} HP:
${row.hp || "-"} Voltage:
${row.voltage || "-"} Phase:
${row.phase || "-"} Discharge:
${row.discharge_size || "-"} Duty:
${row.gpm || "-"} GPM / ${row.head_ft || "-"} FT

Application: ${row.application || "-"}

Verify: ${row.verify_notes || "Final pump selection must be verified before quoting."}

Recommended equivalent for review. Final selection must verify flow, head, voltage, phase, discharge size, solids handling, material, and application requirements.

`; card.querySelector(".quote-btn").addEventListener("click", () => openQuote(row)); results.appendChild(card); }); } function openQuote(row) { const competitor = `${row.competitor_brand || ""} ${row.competitor_model || ""}`.trim(); selectedCompetitor.value = competitor; selectedBarmesa.value = row.barmesa_model || ""; selectedCrossover.value = JSON.stringify(row); formStatus.textContent = ""; submitQuote.disabled = false; submitQuote.textContent = "Send Quote Request"; quoteForm.reset(); selectedCompetitor.value = competitor; selectedBarmesa.value = row.barmesa_model || ""; selectedCrossover.value = JSON.stringify(row); quotePanel.classList.remove("hidden"); } closeQuote.addEventListener("click", () => quotePanel.classList.add("hidden")); quoteForm.addEventListener("submit", async (event) => { event.preventDefault(); submitQuote.disabled = true; submitQuote.textContent = "Sending..."; formStatus.textContent = ""; try { const response = await fetch(quoteForm.action, { method: "POST", body: new FormData(quoteForm), headers: { "Accept": "application/json" } }); if (response.ok) { formStatus.textContent = "Quote request sent. You can submit another request."; quoteForm.reset(); selectedCrossover.value = ""; selectedBarmesa.value = ""; selectedCompetitor.value = ""; submitQuote.textContent = "Send Another Quote Request"; submitQuote.disabled = false; } else { throw new Error("Formspree response was not OK"); } } catch (error) { formStatus.textContent = "Request did not send. Please email sales@northcoastprocessequipment.com."; submitQuote.textContent = "Try Again"; submitQuote.disabled = false; } }); [searchInput, brandFilter, typeFilter, confidenceFilter].forEach(el => { el.addEventListener("input", render); el.addEventListener("change", render); }); clearBtn.addEventListener("click", () => { searchInput.value = ""; brandFilter.value = ""; typeFilter.value = ""; confidenceFilter.value = ""; render(); }); async function init() { try { const response = await fetch(DATA_URL, { cache: "no-store" }); if (!response.ok) throw new Error("CSV not found"); allRows = parseCSV(await response.text()); fillFilters(); render(); } catch (error) { resultCount.textContent = "Could not load crossover database."; results.innerHTML = `

Database not loaded

`; } } init();
